-- Fast Attack Toggle
local fastAttackEnabled = false
local autoFarmEnabled = false
local xp = 0
local level = 1

local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()

-- Quest Related
local currentQuest = nil  -- Tracks the current quest
local npc = game.Workspace.NPCs -- Change this path to where NPCs are in your game

-- Function to accept quests automatically
local function acceptQuest()
    for _, npc in pairs(npc:GetChildren()) do
        if npc:FindFirstChild("QuestGiver") then  -- Ensure NPC has a quest giver script or tag
            -- Interact with NPC to accept the quest
            firetouchinterest(character.HumanoidRootPart, npc.QuestGiver.Part, 0)
            firetouchinterest(character.HumanoidRootPart, npc.QuestGiver.Part, 1)
            print("Quest Accepted: " .. npc.Name)
            currentQuest = npc.Name  -- Track the current quest
        end
    end
end

-- Function to check and complete the quest (if objectives met)
local function completeQuest()
    if currentQuest then
        -- Check if the quest objectives are completed, for example, killing certain enemies
        if currentQuest == "EnemyKillQuest" then  -- Change this to the actual quest name
            local killedEnemies = 0  -- Track how many enemies you have killed for this quest
            -- Ensure the NPC has a quest completion trigger
            if killedEnemies >= 10 then  -- Example condition
                -- Complete the quest by interacting with the NPC again
                firetouchinterest(character.HumanoidRootPart, npc.QuestGiver.Part, 0)
                firetouchinterest(character.HumanoidRootPart, npc.QuestGiver.Part, 1)
                print("Quest Completed: " .. currentQuest)
                currentQuest = nil  -- Reset the quest
            end
        end
    end
end

-- Find Nearest Enemy Function
local function findNearestEnemy()
    local nearestEnemy = nil
    local shortestDistance = math.huge
    for _, enemy in pairs(game.Workspace.Enemies:GetChildren()) do
        if enemy:FindFirstChild("HumanoidRootPart") and enemy:FindFirstChild("Humanoid") then
            local distance = (character.HumanoidRootPart.Position - enemy.HumanoidRootPart.Position).Magnitude
            if distance < shortestDistance then
                shortestDistance = distance
                nearestEnemy = enemy
            end
        end
    end
    return nearestEnemy
end

-- Fast Attack Function
local function fastAttack()
    while fastAttackEnabled do
        wait(0.1)  -- Faster attack speed
        local target = findNearestEnemy()
        if target then
            character.HumanoidRootPart.CFrame = target.HumanoidRootPart.CFrame
            game.ReplicatedStorage:WaitForChild("RemoteEvent"):FireServer("Attack", target)
            xp = xp + 10  -- Gain XP for each attack (Example value)
            if xp >= level * 100 then  -- Level Up Condition
                level = level + 1
                xp = 0
            end
            xpLabel.Text = "XP: " .. xp .. " | Level: " .. level
        end
        completeQuest()  -- Check if quest needs completion
    end
end
